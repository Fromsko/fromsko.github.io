```sql
-- 查询员工表中的数据
SELECT * FROM EMP;
-- 查询部门表的数据
SELECT * FROM DEPT;
-- select 执行顺序(from, where, group by having, order by)
-- 查询年薪超过 2K 的员工
SELECT ENAME,
       SAL,
       SAL * 12 "YEAR_SAL"
       FROM EMP
       WHERE SAL > 2000
       ORDER BY YEAR_SAL;
T
-- GROUP BY 分组
-- 查询各个部门的最高薪水
SELECT ENAME,
       SAL,
       DEPTNO
       FROM EMP
       WHERE (DEPTNO, SAL) IN ( SELECT DEPTNO,
       MAX(SAL)
       FROM EMP
       WHERE DEPTNO IS NOT NULL
       GROUP BY DEPTNO);

-- 表连接查询
SELECT * FROM EMP JOIN SELECT DEPTNO,
       MAX(SAL)
       FROM EMP
       GROUP BY DEPTNO;
```

```plsql
SELECT
    ENAME,
    JOB,
    DNAME,
    LOCATION
FROM
    EMP_WHC  E
    JOIN DEPT_WHC D
    ON E.DEPTNO=D.DEPTNO
WHERE
    D.LOCATION ='张家界';
```
